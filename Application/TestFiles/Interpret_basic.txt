# Przykład pliku źródłowego 

int nwd(int a, int b)
{
    if(a==b)
    {
        return a;
    }

    if (a > b)
    {
        return nwd(a-b, b); # wywołania rekurencyjne
    }

    return nwd(b-a, a);
}

int factorial(int a)
{
    if(a == 1)
        return 1;

    return a * factorial(a-1);
}

void main()
{	
    int x = 40;
    int y = 30;
    int z = 5;

    print("nwd (", x, ", ", y, ") = ", nwd(x, y)); 
    print("Silnia ", z, " wynosi = ", factorial(z));  # Silnia 3 = 6
}
